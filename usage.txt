1. Train ViT model

cd experiments/models
python3 train_vit.py --config cifar-config.json --dataset CIFAR10 --batch_size 64 --epochs 100 --lr 0.01 --save_every 25 --device cuda:0 --exp_name cifar10_experiment --base_dir ../..
python3 train_vit.py --config mnist-config.json --dataset MNIST --batch_size 64 --epochs 20 --lr 0.01 --save_every 5 --device cuda:0 --exp_name mnist_experiment --base_dir ../..

2. Convert images and texts in the expected format

cd data-preprocessing
bash main.sh

3. Prepare experiments
cd experiments
python3 prepare_experiment.py --default

4a. For a single exploration:
python3 vit_exploration.py --experiment-path ./experiments_data/cifar-1-50-all --out-dir ../res
python3 bert_exploration.py --experiment-path ./experiments_data/winobias-1p0-32-target-word --out-dir ../res

4b. For a single interpretation:
python3 vit_exploration_interpretation.py --experiment-path experiments_data/cifar-1-50-all --results-dir ../res/cifar-1-50-all-20250403-200221

4c. For all explorations and interpretation:
cd sh-scripts
bash image-experiments.sh
